# KORAL Infrastructure — Tool Requirements
# Member 2 (DevOps) — these are the required versions for the infrastructure layer
#
# This is NOT a Python requirements file.
# It documents the exact tool versions needed to run, deploy, and operate KORAL.
# Install each tool before running any script in scripts/
#
# ─────────────────────────────────────────────
# REQUIRED TOOLS
# ─────────────────────────────────────────────

# Kubernetes local cluster
minikube >= 1.32.0
# Install: https://minikube.sigs.k8s.io/docs/start/
# Windows: winget install Kubernetes.minikube

# Kubernetes CLI
kubectl >= 1.28.0
# Install: https://kubernetes.io/docs/tasks/tools/
# Windows: winget install Kubernetes.kubectl

# Helm — Kubernetes package manager
helm >= 3.12.0
# Install: https://helm.sh/docs/intro/install/
# Windows: winget install Helm.Helm

# Container runtime
docker >= 24.0.0
# Install: https://docs.docker.com/get-docker/
# Windows: Docker Desktop

# ─────────────────────────────────────────────
# HELM CHART DEPENDENCIES (installed by bootstrap.sh)
# ─────────────────────────────────────────────

# Repo: https://prometheus-community.github.io/helm-charts

# Includes: Prometheus, Alertmanager, Grafana, cAdvisor, node_exporter

# Fluentd log collector
# Repo: https://fluent.github.io/helm-charts
# Chart: fluentd >= 0.5.0
